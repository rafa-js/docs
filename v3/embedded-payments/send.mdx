---
title: Send
description: 'Trigger a payment from a connected user'
---

```typescript
import { getInstance, Connect } from '@handcash/sdk';

const sdk = getInstance({ appId: 'YOUR_APP_ID', appSecret: 'YOUR_APP_SECRET' });
const client = sdk.getAccountClient('<AUTH_TOKEN>');

await Connect.pay({
  client,
  body: {
    instrumentCurrencyCode: 'BSV',
    denominationCurrencyCode: 'USD',
    receivers: [{ destination: 'nosetwo', sendAmount: 0.01 }],
  },
});
```
